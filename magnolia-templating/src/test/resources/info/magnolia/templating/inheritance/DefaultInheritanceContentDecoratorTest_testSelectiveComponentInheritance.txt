
# Tests that a property 'inherited' on components control whether they're inherited or not


[Input]
/page1.@type = mgnl:page
/page1/main.@type = mgnl:area

/page1/main/0.@type = mgnl:component
/page1/main/0.inherited = false
/page1/main/1.@type = mgnl:component
/page1/main/1.inherited = true

/page1/page2.@type = mgnl:page
/page1/page2/main.@type = mgnl:area
/page1/page2/main/0.@type = mgnl:component
/page1/page2/main/0.title = Title


[Expected]
/page1.@type = mgnl:page
/page1/main.@type = mgnl:area

/page1/main/0.@type = mgnl:component
/page1/main/0.inherited = false
/page1/main/1.@type = mgnl:component
/page1/main/1.inherited = true

/page1/page2.@type = mgnl:page
/page1/page2/main.@type = mgnl:area
/page1/page2/main/1.@type = mgnl:component
/page1/page2/main/1.inherited = true
/page1/page2/main/0.@type = mgnl:component
/page1/page2/main/0.title = Title
